Summary:

You can format browser url depending on your (SEO) needs by your own: 

serializeStateToUrl: function (state, queryString) {
  const search = queryString.stringify({
    ...state.filter,
    ...state.sort,
  });

  return location.protocol + '//' + location.host + location.pathname
    + (search ? ('?' + search) : '');
},

Then because of you serialize url by your own you need to deserialize it by your own too: 

deserializeUrlToState: function (queryLib, normalizeValues) {
  const query = queryString.parse(location.search.replace(/^\?/, ''));

  return {
    filter: normalizeValues(query) || {},
    sort: normalizeValues(query) || {},
    pagination: {
      page: parseInt(query['p']) || null,
      perPage: parseInt(query['pp']) || null
    }
  }
},
